            const fpvIframe = document.getElementById('fpv-iframe');
            if (fpvIframe && fpvIframe.contentWindow) {
                fpvIframe.contentWindow.postMessage({ type: 'KEY_DOWN', key: e.key, code: e.code }, '*');
            }
            const mapIframe = document.getElementById('map-iframe');
            if (mapIframe && mapIframe.contentWindow) {
                mapIframe.contentWindow.postMessage({ type: 'KEY_DOWN', key: e.key, code: e.code }, '*');
            }
        });

        document.addEventListener('keyup', (e) => {
            if (document.activeElement && document.activeElement.tagName === 'INPUT') return;
            const fpvIframe = document.getElementById('fpv-iframe');
            if (fpvIframe && fpvIframe.contentWindow) {
                fpvIframe.contentWindow.postMessage({ type: 'KEY_UP', key: e.key, code: e.code }, '*');
            }
            const mapIframe = document.getElementById('map-iframe');
            if (mapIframe && mapIframe.contentWindow) {
                mapIframe.contentWindow.postMessage({ type: 'KEY_UP', key: e.key, code: e.code }, '*');
            }
        });
        
        let cachedInitEntities = null;

        // --- MASTER MESSAGE ROUTER ---
        // Bounces messages between the decoupled IFrames
        window.addEventListener('message', (e) => {
           if (!e.data || !e.data.type) return;

           // When Map Engine spins up, immediately send it the cached dungeon data
           if (e.data.type === 'MAP_ENGINE_READY' && cachedInitEntities) {
               const mapFrame = document.getElementById('map-iframe');
               if (mapFrame && mapFrame.contentWindow) {
                   mapFrame.contentWindow.postMessage(cachedInitEntities, '*');
               }
           }

           // Route movement data to the Map Engine to keep the Map Camera synced
           if (e.data.type === 'PLAYER_MOVE') {
               const mapFrame = document.getElementById('map-iframe');
               if (mapFrame && mapFrame.contentWindow) {
                   mapFrame.contentWindow.postMessage(e.data, '*');
               }
           }
           
           if (e.data.type === 'PLAYER_MOVE_STATE') {
               window.parent.postMessage(e.data, '*');
               
               // Fade HUD when moving logic disabled to prevent flashing panels
           }
           
           if (e.data.type === 'PLAYER_IDLE') {
               // Player stopped moving map/rendering.
               // We catch the `isMoving: false` in PLAYER_MOVE_STATE instead for robustness.
           }
           
           // Combat HUD visibility
           if (e.data.type === 'SHOW_COMBAT') {
               const combatWrapper = document.getElementById('combat-panels-wrapper');
               if (combatWrapper) combatWrapper.classList.add('active');
               const cf = document.getElementById('combat-iframe');
               if (cf && cf.contentWindow) cf.contentWindow.postMessage(e.data, '*');
           }
           
           if (e.data.type === 'HIDE_COMBAT' || e.data.type === 'HIDE_ALL') {
               const combatWrapper = document.getElementById('combat-panels-wrapper');
               if (combatWrapper) combatWrapper.classList.remove('active');
               const cf = document.getElementById('combat-iframe');
               if (cf && cf.contentWindow) cf.contentWindow.postMessage(e.data, '*');
           }
           
           // Route the generated dungeon layout from the FPV engine to the Map engine and AI
           if (e.data.type === 'INIT_ENTITIES') {
               cachedInitEntities = e.data;
               const mapFrame = document.getElementById('map-iframe');
               if (mapFrame && mapFrame.contentWindow) {
                   mapFrame.contentWindow.postMessage(e.data, '*');
               }
               // Note: The AI frame router will also be placed here eventually
           }
           
           // Route Virtual Keypad events to engines
           if (e.data.type === 'KEY_DOWN' || e.data.type === 'KEY_UP') {
               const fpvFrame = document.getElementById('fpv-iframe');
               if (fpvFrame && fpvFrame.contentWindow) {
                   fpvFrame.contentWindow.postMessage(e.data, '*');
               }
               const mapFrame = document.getElementById('map-iframe');
               if (mapFrame && mapFrame.contentWindow) {
                   mapFrame.contentWindow.postMessage(e.data, '*');
               }
           }
           
           // Route UI Actions from Panels (relay to original FPV engine)
           if (['EXPLORE_ROOM', 'SEARCH_ADJACENT', 'AUTO_EXPLORE', 'APPROACH_MONSTER', 'TEXT_COMMAND', 'FPV_ACTION'].includes(e.data.type)) {
               const fpvFrame = document.getElementById('fpv-iframe');
               if (fpvFrame && fpvFrame.contentWindow) {
                   fpvFrame.contentWindow.postMessage(e.data, '*');
               }
           }
        });
      });
    </script>
